JavaScript 轉換型別小技巧

var number = 1;
var string = '1';
var bool = true;

string === number; // false
bool === number; // false
bool === string; // false

string 與 number 轉換

string === '' + number; // true
+string === number; // true

boolean 與 number 轉換

+bool === number; // true  
bool === !!number // true

boolean 與string 轉換

bool === !!string; // true
'' + bool === string; // true
